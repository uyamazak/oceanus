(

    TIMESTAMP(delete_date) as delete_date_ts
  FROM
  [oceanus-dev:bizocean.member_del]
)
"""

exit_member_period = bq.Query("""
SELECT
  STRFTIME_UTC_USEC(delete_date_ts, "%m/%d-%a") as delete_date,
  count(member_id) as count,
  AVG(period_day) as average,
  NTH(501, QUANTILES(period_day, 1001)) as median,
  MAX(period_day) as max,
FROM
  (SELECT
    member_id,
    DATEDIFF(delete_date_ts, create_date_ts ) as period_day,
    delete_date_ts
  FROM
    (SELECT
      member_id,
      create_date_ts,
      delete_date_ts
    FROM
      {BIZOCEAN_MEMBER_DEL} as bim_del

      delete_date_ts >= TIMESTAMP("{A_MONTH_AGO}")
      AND
      delete_date_ts <= TIMESTAMP("{TODAY}")
    )
  )
GROUP BY delete_date
ORDER BY delete_date
""".format(BIZOCEAN_MEMBER=BIZOCEAN_MEMBER,
           BIZOCEAN_MEMBER_DEL=BIZOCEAN_MEMBER_DEL,

# bizocean memberテーブル
BIZOCEAN_MEMBER = """
(
SELECT
  member_id,
  create_date,
  TIMESTAMP(create_date)  as create_date_ts,
  INTEGER(age) as age
FROM
[oceanus-dev:bizocean.member]
)
"""

# bizocean member_delテーブル
BIZOCEAN_MEMBER_DEL = """
(
SELECT
  member_id, 
  create_date,
  TIMESTAMP(create_date) as create_date_ts,
  INTEGER(age) as age
FROM
[oceanus-dev:bizocean.member_del]
)
"""

AGE_TAMPLATE = """
SELECT
  date,
  count(*) as count,
  age
FROM
  (
  SELECT
    member_id,
    create_date,
    create_date_ts,
    INTEGER(FLOOR(INTEGER(age)/10)) * 10 as age,
    STRFTIME_UTC_USEC(create_date_ts, "%m/%d-%a") as date
  FROM
    {BIZOCEAN_MEMBER} as bim,
    {BIZOCEAN_MEMBER_DEL} as bim_del
  WHERE
    create_date_ts >= TIMESTAMP("{A_MONTH_AGO}")
    AND
    create_date_ts <= TIMESTAMP("{TODAY}")
    AND
    age BETWEEN {{AGE_FROM}} AND {{AGE_TO}}
  )
GROUP BY date, age
""".format(A_MONTH_AGO=A_MONTH_AGO,
           TODAY=TODAY,
           BIZOCEAN_MEMBER=BIZOCEAN_MEMBER,
           BIZOCEAN_MEMBER_DEL=BIZOCEAN_MEMBER_DEL
          )
AGE_TAMPLATE

AGE_20 = AGE_TAMPLATE.format(AGE_FROM=20, AGE_TO=29)
AGE_30 = AGE_TAMPLATE.format(AGE_FROM=30, AGE_TO=39)
AGE_40 = AGE_TAMPLATE.format(AGE_FROM=40, AGE_TO=49)
AGE_50 = AGE_TAMPLATE.format(AGE_FROM=50, AGE_TO=59)
AGE_60 = AGE_TAMPLATE.format(AGE_FROM=60, AGE_TO=69)
#print(AGE_20)

new_member_age = bq.Query("""
SELECT
  age_20.date as date,
  age_20.count as age_20,
  age_30.count as age_30,
  age_40.count as age_40,
  age_50.count as age_50,
  age_60.count as age_60
FROM
  ({AGE_20}) as age_20
  JOIN
  ({AGE_30}) as age_30
  ON age_20.date == age_30.date
  JOIN
  ({AGE_40}) as age_40
  ON age_20.date == age_40.date
  JOIN
  ({AGE_50}) as age_50
  ON age_20.date == age_50.date
  JOIN
  ({AGE_60}) as age_60
  ON age_20.date == age_60.date
ORDER BY date
""".format(BIZOCEAN_MEMBER=BIZOCEAN_MEMBER,
           BIZOCEAN_MEMBER_DEL=BIZOCEAN_MEMBER_DEL,
           TODAY=TODAY,
           YESTERDAY=YESTERDAY,
           A_MONTH_AGO=A_MONTH_AGO,
           AGE_20=AGE_20,
           AGE_30=AGE_30,
           AGE_40=AGE_40,
           AGE_50=AGE_50,
           AGE_60=AGE_60)).results()
